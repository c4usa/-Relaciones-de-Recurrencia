void invertirCad(char cadena[], int inicio, int fin)
{
    if (inicio >= fin)
    {
        return;
    }
    char temporal = cadena[inicio];
    cadena[inicio] = cadena[fin];
    cadena[fin] = temporal;
    invertirCad(cadena, inicio + 1, fin - 1);
}

// Funci贸n para calcular el factorial
int factorial(int n)
{
    if (n <= 1)
    {
        return 1;
    }
    return n * factorial(n - 1);
}
